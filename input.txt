import ioPlus;
class ConstFold {
    public int fun(int b) {
        int a;
        a = 10;
        return a;
    }

	public static void main(String[] args) {
        ConstFold k;
        int a;
        int b;
        k = new ConstFold();
        a = 10;
        b = 20;
	    ioPlus.printResult(new ConstFold().fun(10) + k.fun(a * b));
	}
}
